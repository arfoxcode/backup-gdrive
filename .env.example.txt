# KONFIGURASI BACKUP GOOGLE DRIVE
# Salin file ini menjadi .env dan isi nilainya
# ─────────────────────────────────────────────

# Path folder/file yang ingin di-backup
# Contoh: ./data  atau  /home/user/dokumen  atau  D:/MyFiles
BACKUP_SOURCE=./data

# Folder temporary untuk menyimpan file zip sementara
BACKUP_TEMP=./temp_backups

# ID Folder Google Drive tujuan backup
# Cara dapat: buka folder di Drive → lihat URL → ambil bagian setelah /folders/
# Contoh URL: https://drive.google.com/drive/folders/1A2B3C4D5E6F7G8H
# Maka ID-nya: 1A2B3C4D5E6F7G8H
# Kosongkan untuk upload ke root Drive
GDRIVE_FOLDER_ID=

# Jadwal backup (format cron)
# ┌──────────── menit (0-59)
# │  ┌─────────── jam (0-23)
# │  │  ┌────────── hari bulan (1-31)
# │  │  │  ┌───────── bulan (1-12)
# │  │  │  │  ┌──────── hari minggu (0-7, 0=Minggu)
# │  │  │  │  │
# 0  2  *  *  *   = setiap hari jam 02:00
# 0  */6 * *  *   = setiap 6 jam
# 0  9  *  *  1   = setiap Senin jam 09:00
# */30 * * *  *   = setiap 30 menit
CRON_SCHEDULE=0 2 * * *

# Jumlah maksimal backup yang disimpan di Google Drive
# Backup paling lama akan otomatis dihapus jika melebihi batas ini
MAX_BACKUPS=7

# Path file credentials Google OAuth2 (unduh dari Google Cloud Console)
CREDENTIALS_PATH=./credentials.json

# Path file token (dibuat otomatis setelah otorisasi pertama)
TOKEN_PATH=./token.json
